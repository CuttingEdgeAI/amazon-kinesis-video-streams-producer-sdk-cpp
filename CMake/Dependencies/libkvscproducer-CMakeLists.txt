cmake_minimum_required(VERSION 2.8)

project(libkvscproducer-download NONE)

include(ExternalProject)

# I'm using my own fork since the current master doesn't detect already built open source libs.
# So unless you want to waste your life building openssl a million times use this fork or get
# it or a better fix merged to the upstream.
ExternalProject_Add(libkvscproducer-download
	GIT_REPOSITORY    https://github.com/vattay/amazon-kinesis-video-streams-producer-c.git
	GIT_TAG           bd281677f26ae73a61f0635828c2ebe207947726
	SOURCE_DIR        "${CMAKE_CURRENT_BINARY_DIR}/kvscproducer-src"
	BINARY_DIR        "${CMAKE_CURRENT_BINARY_DIR}/kvscproducer-build"
	CONFIGURE_COMMAND ""
	BUILD_COMMAND     ""
	INSTALL_COMMAND   ""
	TEST_COMMAND      ""
)
